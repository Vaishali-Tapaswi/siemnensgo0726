Lab6 - Channel
	package main

import (
	"fmt"
	"time"
	"strconv"
)
func sender(ch chan string){
	defer close(ch)
	for i := 1; i <= 5; i++ {
		fmt.Println("---in count " ,i) 
		ch <- "Simple"  + strconv.Itoa(i);
		time.Sleep(time.Millisecond * 100)
	}
	
}

func rec (ch chan string) {
	fmt.Println("Getting data from channel")
	for msg := range ch { 
			fmt.Println("Got ", msg)
	}
}
func main(){
	ch := make(chan string,1)
	fmt.Println("channel created")
	go rec(ch)
	go sender(ch)
	time.Sleep(15 * time.Second)
}
